/*
2829. Determine the Minimum Sum of a k-avoiding Array
*/
class Solution {
public:
    int minimumSum(int n, int k) 
    {
        unordered_set<int>s;
        int c=1;
        for(int i=1;s.size()<n;i++)        
        {
            if(s.find(k-i)==s.end())
            s.insert(i);
        }
        int sum=0;
        for(auto it:s)
        {
            sum+=it;
        }
        return sum;
  
    }
};
